import React from 'react';
import './App.css';

//Clase
class Helloworld extends React.Component{
    state ={
        show: true
    }
    toggleShow=()=>{
        this.setState({show: !this.state.show})
    }

    render() {
        if (this.state.show){
            return(
                <div id="Hello">
                    <h3>{this.props.mytext}</h3>
                    {this.props.subtitle}
                    <button onClick={this.toggleShow}>Toggle Show</button>
                </div>
            )
        }else {
            return <h1>"There are not elements"
            <button onClick={this.toggleShow}>
                Toggle Show
            </button>
            </h1>
        }
    }
}


function App() {
    return (
        <div> this is my componet:
            <Helloworld mytext="Hola mundo" subtitle="Otro texto"/>
            <Helloworld mytext="Segundo" subtitle="2 texto"/>
            <Helloworld mytext="tercero" subtitle="3 texto"/>
        </div>
    );
}



export default App;
